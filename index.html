import cv2 import mediapipe as mp import numpy as np import math mp_hands = mp.solutions.hands hands = mp_hands.Hands( max_num_hands=1, min_detection_confidence=0.7, min_tracking_confidence=0.7 ) mp_draw = mp.solutions.drawing_utils # 计算两点距离 def distance(p1, p2): return math.hypot(p2[0]-p1[0], p2[1]-p1[1]) def get_gesture(lm): thumb = lm[4] index = lm[8] middle = lm[12] ring = lm[16] pinky = lm[20] wrist = lm[0] # 手指是否伸展 fingers = [] fingers.append(1 if lm[4][0] > lm[3][0] else 0) # 拇指 fingers.append(1 if lm[8][1] < lm[6][1] else 0) fingers.append(1 if lm[12][1] < lm[10][1] else 0) fingers.append(1 if lm[16][1] < lm[14][1] else 0) fingers.append(1 if lm[20][1] < lm[18][1] else 0) # 👍 if fingers == [1,0,0,0,0]: return "THUMBS UP" # ✋ 张开手 if fingers == [1,1,1,1,1]: return "OPEN HAND" # ✊ 拳头 if fingers == [0,0,0,0,0]: return "FIST" # ✌️ if fingers == [0,1,1,0,0]: return "PEACE" # 👉 食指 if fingers == [0,1,0,0,0]: return "POINT" # 🤏 捏合（简单判断：食指+拇指距离很近） d = distance(index, thumb) if d < 40: return "PINCH" return "UNKNOWN" cap = cv2.VideoCapture(0) while True: ret, frame = cap.read() if not ret: break frame = cv2.flip(frame, 1) rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB) result = hands.process(rgb) gesture = "NO HAND" if result.multi_hand_landmarks: for handLms in result.multi_hand_landmarks: lm_list = [] h, w, _ = frame.shape for id, lm in enumerate(handLms.landmark): lm_list.append((int(lm.x*w), int(lm.y*h))) mp_draw.draw_landmarks(frame, handLms, mp_hands.HAND_CONNECTIONS) gesture = get_gesture(lm_list) # ====== 科技UI ====== overlay = frame.copy() cv2.rectangle(overlay, (20, 20), (300, 120), (0, 255, 255), -1) alpha = 0.3 frame = cv2.addWeighted(overlay, alpha, frame, 1-alpha, 0) cv2.putText(frame, "NEXUS GESTURE AI", (30, 55), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0,0,0), 2) cv2.putText(frame, f"GESTURE: {gesture}", (30, 95), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255,255,255), 2) # 发光效果模拟 cv2.circle(frame, (400, 80), 10, (0,255,255), -1) cv2.imshow("NEXUS AI - GESTURE SYSTEM", frame) if cv2.waitKey(1) & 0xFF == ord('q'): break cap.release() cv2.destroyAllWindows()
